packages = ["matplotlib", "numpy"] import numpy as np import matplotlib.pyplot as plt x1 = 2 def f(x): return x**2 # Define parabola derivative def slope(x): return 2*x def g(x): return 2*x1*x+(x1**2-2*x1*x1) # Define x data range for parabola x = np.linspace(-5,5,100) plt.plot(x, f(x)) plt.plot(x, g(x)) plt.show()
